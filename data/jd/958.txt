formSubmittedOk = true;
try { sessionStorage.setItem('apply_submitted', '1'); } catch (e) {}
window.location = '/apply/confirmation/';
} else {
var msg = (result.body && result.body.error) || 'Something went wrong. Please try again, or email careers@mechanize.work.';
track('apply_form_submit_failed', { reason: 'server', status: result.status, error_message: msg, roles: roles });
showMessage(msg, 'error');
submitBtn.disabled = false;
submitBtn.textContent = 'Submit application';
}
})
.catch(function () {
track('apply_form_submit_failed', { reason: 'network', roles: roles });
showMessage('Network error. Please try again, or email careers@mechanize.work.', 'error');
submitBtn.disabled = false;
submitBtn.textContent = 'Submit application';
});
});
})();
